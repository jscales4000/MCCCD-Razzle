# MCCCD-AA140 Touchpanel Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a CH5-Svelte touchpanel + SIMPL# Pro control system for Maricopa Community Colleges District room AA140 — three displays with per-display source pickers + one-shot mirror to D3, three 1Beyond cameras with PTZ + VTC ingest + tracking modes, Q-SYS-driven audio, and PoE occupancy automation.

**Architecture:** Single ch5-svelte-v2 panel project (Svelte 5 + Vite 6 + TypeScript) deployed identically to TS-1070, TSW-1070, and WebXPanel. SIMPL# Pro on RMC4 owns NVX routing, Q-SYS PA-module audio control, 1Beyond REST control, and the occupancy state machine. The `.cce` contract is the source of truth for the panel↔processor signal interface; `src/lib/contract.ts` is the hand-maintained TypeScript mirror.

**Tech Stack:** Svelte 5, Vite 6, TypeScript 5, `@crestron/ch5-crcomlib` 2.17.x, `@crestron/ch5-webxpanel` 2.8.x, `@crestron/ch5-shell-utilities-cli` 2.17.x; SIMPL# Pro on 4-Series, Crestron Q-SYS PA Module, 1Beyond HTTP REST API.

**Spec:** [`docs/superpowers/specs/2026-04-26-mcccd-aa140-design.md`](../specs/2026-04-26-mcccd-aa140-design.md)

```svelte
<script lang="ts">
  import { CrComLib } from '../lib/CrComLib';
  import { SIGNALS } from '../lib/contract';
  import { nvxAutoSwitchSrc } from '../lib/stores/signals';

  interface Props {
    label: string;             // "Display 1" | "Display 2" | "Display 3"
    sourceSetSignal: string;   // SIGNALS.display1Source / display2Source / display3Source
    activeSourceFb: number;    // current source from feedback store (1..5, 0 = none)
    audioActive?: boolean;     // optional, only D1/D2
    onAudioToggle?: () => void; // optional, only D1/D2
    onMirrorToD3?: () => void;  // optional, only D1/D2
  }

  let { label, sourceSetSignal, activeSourceFb, audioActive = false, onAudioToggle, onMirrorToD3 }: Props = $props();

  const sources = [
    { label: 'Room PC', value: 1 },
    { label: 'Ext PC', value: 2 },
    { label: 'AirMedia', value: 3 },
    { label: 'HDMI', value: 4 },
    { label: 'USB-C', value: 5 },
  ];

  function selectSource(value: number) {
    CrComLib.publishEvent('n', sourceSetSignal, value);
  }

  function activeLabel(): string {
    const s = sources.find(s => s.value === activeSourceFb);
    return s ? s.label : '—';
  }

  // Show NVX-384 auto-switch badge when user picked HDMI but USB-C is active (or vice versa)
  function autoSwitchBadge(): string | null {
    if (activeSourceFb !== 4 && activeSourceFb !== 5) return null;
    const actual = $nvxAutoSwitchSrc; // 1=HDMI, 2=USB-C
    if (activeSourceFb === 4 && actual === 2) return 'USB-C active';
    if (activeSourceFb === 5 && actual === 1) return 'HDMI active';
    return null;
  }
</script>

<div class="glass-card display-tile" class:audio-active={audioActive}>
  <div class="tile-header">
    <div class="tile-meta">
      <p class="tile-label">{label}</p>
      <p class="tile-source">{activeLabel()}</p>
      {#if autoSwitchBadge()}
        <span class="autoswitch-badge">↳ {autoSwitchBadge()}</span>
      {/if}
    </div>
    <div class="tile-actions">
      {#if onAudioToggle}
        <button
          class="btn icon-btn"
          class:active={audioActive}
          onclick={onAudioToggle}
```

```csharp
using Crestron.SimplSharp;
using Crestron.SimplSharpPro;
using Crestron.SimplSharpPro.DM.Streaming;
using MCCCD_AA140.Generated;

namespace MCCCD_AA140
{
    public class NvxRoutingService
    {
        // IPID map per spec
        private const uint IPID_E30_ROOM_PC   = 0x11;
        private const uint IPID_E30_EXT_PC    = 0x12;
        private const uint IPID_E30_AIRMEDIA  = 0x13;
        private const uint IPID_NVX_384       = 0x14; // HDMI + USB-C combo
        private const uint IPID_D200_DISP1    = 0x21;
        private const uint IPID_D200_DISP2    = 0x22;
        private const uint IPID_D200_DISP3    = 0x23;

        private readonly MainContract _c;
        private readonly CrestronControlSystem _cs;

        // Hardware refs
        private DmNvx351   _encRoomPc;       // E30 ≈ DmNvx351 family — adjust to actual class per E30 model
        private DmNvx351   _encExtPc;
        private DmNvx351   _encAirMedia;
        private DmNvx384   _encHdmiUsbc;     // 4-input encoder
        private DmNvxD30   _decDisp1;        // D200 ≈ DmNvxD30 family — adjust to actual class
        private DmNvxD30   _decDisp2;
        private DmNvxD30   _decDisp3;

        // Source index → encoder stream URL map (fed by encoder OnlineStatusChange)
        private string[]   _sourceStreamUrls = new string[6]; // index 1..5

        public NvxRoutingService(MainContract c, CrestronControlSystem cs)
        {
            _c = c;
            _cs = cs;
        }

        public void Initialize()
        {
            // Construct + register encoders
            _encRoomPc   = new DmNvx351(IPID_E30_ROOM_PC,  _cs);
            _encExtPc    = new DmNvx351(IPID_E30_EXT_PC,   _cs);
            _encAirMedia = new DmNvx351(IPID_E30_AIRMEDIA, _cs);
            _encHdmiUsbc = new DmNvx384(IPID_NVX_384,      _cs);

            _decDisp1 = new DmNvxD30(IPID_D200_DISP1, _cs);
            _decDisp2 = new DmNvxD30(IPID_D200_DISP2, _cs);
            _decDisp3 = new DmNvxD30(IPID_D200_DISP3, _cs);

            foreach (var dev in new GenericBase[] { _encRoomPc, _encExtPc, _encAirMedia, _encHdmiUsbc, _decDisp1, _decDisp2, _decDisp3 })
                dev.Register();

            // Wire NVX-384 active-input feedback for the autoswitch badge
            _encHdmiUsbc.BaseEvent += (dev, args) => {
                // SIG_TYPE active source feedback: 1=HDMI, 2=USB-C — exact event id depends on Crestron SDK version
                _c.NvxAutoSwitchSrc = (ushort)CurrentNvx384ActiveInput();
            };

            // Wire panel commands → NVX routes
            _c.Display1Source.OnAnalogChange += (v) => RouteSourceToDisplay(v, 1);
            _c.Display2Source.OnAnalogChange += (v) => RouteSourceToDisplay(v, 2);
            _c.Display3Source.OnAnalogChange += (v) => RouteSourceToDisplay(v, 3);

            // Mirror pulses
            _c.D1MirrorToD3.OnDigitalRise += () => MirrorTo3(_c.Display1SourceFb);
            _c.D2MirrorToD3.OnDigitalRise += () => MirrorTo3(_c.Display2SourceFb);
        }

        public void RouteSourceToDisplay(ushort srcIndex, int displayNum)
        {
            var url = _sourceStreamUrls[srcIndex];
            if (string.IsNullOrEmpty(url)) {
                ErrorLog.Warn("NVX: no stream URL for source {0}", srcIndex);
                return;
            }
            DmNvxD30 dec = displayNum switch { 1 => _decDisp1, 2 => _decDisp2, 3 => _decDisp3, _ => null };
            if (dec == null) return;
            dec.Control.ServerUrl.StringValue = url;
            // Feedback
            switch (displayNum) {
                case 1: _c.Display1SourceFb = srcIndex; break;
                case 2: _c.Display2SourceFb = srcIndex; break;
                case 3: _c.Display3SourceFb = srcIndex; break;
            }
        }

        public void MirrorTo3(ushort srcFromD1OrD2)
        {
            if (srcFromD1OrD2 == 0) return; // nothing to mirror
            RouteSourceToDisplay(srcFromD1OrD2, 3);
        }

        private int CurrentNvx384ActiveInput()
        {
            // Read NVX-384 currently-active physical input from device feedback.
```

```csharp
            var url = $"http://{_camIps[_active]}/cgi-bin/ptz?action=stop";
            HttpFireAndForget(url);
        }

        private void RecallPreset(ushort idx) {
            var url = $"http://{_camIps[_active]}/cgi-bin/preset?action=recall&id={idx}";
            HttpFireAndForget(url);
        }
        private void SavePreset(ushort idx) {
            var url = $"http://{_camIps[_active]}/cgi-bin/preset?action=save&id={idx}";
            HttpFireAndForget(url);

            HttpFireAndForget(url);
```

```csharp
            // Initial state: vacant

        }

        // Call from sensor event (or simulate via SIMPL# Debug Tool)
        public void HandleOccupancyTransition(bool nowOccupied)
        {
            if (nowOccupied && !_occupied) {
                // Empty → Occupied
                _occupied = true;
                CancelShutdownTimer();
                _c.OccupancyState = 1;
                _c.ShutdownCountdown = 0;
                OnOccupiedTransition?.Invoke();
            }
            else if (!nowOccupied && _occupied) {
                // Occupied → Empty: start shutdown timer
                _occupied = false;
                StartShutdownTimer();
            }
        }

        private void StartShutdownTimer()
        {
            _minutesRemaining = SHUTDOWN_DELAY_MIN;
            _c.OccupancyState = 2;
            _c.ShutdownCountdown = (ushort)_minutesRemaining;
            _shutdownTimer = new CTimer(TimerTick, null, 60000, 60000); // every minute
        }

        private void CancelShutdownTimer()
        {
            _shutdownTimer?.Stop();
            _shutdownTimer?.Dispose();
            _shutdownTimer = null;
        }

        private void TimerTick(object _)
        {
            _minutesRemaining--;
            _c.ShutdownCountdown = (ushort)System.Math.Max(0, _minutesRemaining);
            if (_minutesRemaining <= 0) {
                CancelShutdownTimer();
                _c.OccupancyState = 0;
                OnSoftShutdownExpired?.Invoke();
            }
        }
    }
```

```csharp
using Crestron.SimplSharp;

namespace MCCCD_AA140

    public class SystemPowerController
    {
        private readonly MainContract _c;
        private readonly NvxRoutingService _nvx;
        private readonly QsysAudioService _audio;
        private readonly CrestronControlSystem _cs;

        private bool _systemOn;

        // Last-known sources before shutdown — restored on power-on
        private ushort _lastD1 = 1;
        private ushort _lastD2 = 1;

        public SystemPowerController(MainContract c, NvxRoutingService nvx, QsysAudioService audio, CrestronControlSystem cs)
        {
            _c = c; _nvx = nvx; _audio = audio; _cs = cs;
        }

        public void Initialize()
        {
            _c.DisplayPower.OnDigitalRise += () => {
                if (_systemOn) PowerDownSequence();
                else            PowerUpSequence();
            };

            // Snapshot last-active sources whenever they change
            _c.Display1SourceFb_Changed += (v) => { if (v != 0) _lastD1 = v; };
            _c.Display2SourceFb_Changed += (v) => { if (v != 0) _lastD2 = v; };
        }

        public void PowerUpSequence()
        {
            ErrorLog.Notice("AA140: PowerUp");
            _systemOn = true;

            // Restore last-active sources to D1, D2
            _nvx.RouteSourceToDisplay(_lastD1, 1);
            _nvx.RouteSourceToDisplay(_lastD2, 2);

            // D3 boot init: one-shot copy from D2
            _nvx.RouteSourceToDisplay(_lastD2, 3);

            // Audio default: follow D1 (UI can change)
            _c.AudioOutputSelect = 1;

            // Unmute
            _c.MuteAll = false;
        }

        public void PowerDownSequence()
        {
            ErrorLog.Notice("AA140: PowerDown");
            _systemOn = false;

            // Mute audio + mics (preserve user mic-mute states? — spec choice; here we mute)
            _c.MuteAll = true;

            // Source 0 = none (NVX decoder receives empty URL in NvxRoutingService)
            _nvx.RouteSourceToDisplay(0, 1);
            _nvx.RouteSourceToDisplay(0, 2);
            _nvx.RouteSourceToDisplay(0, 3);
        }
    }
}
```
